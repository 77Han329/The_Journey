class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        res = []
        path = []

        res.append(path)

        def backtracking(nums, start_idx):

            if start_idx >=len(nums):
                return 
            
            for i in range(start_idx,len(nums)):
                path.append(nums[i])
                res.append(path[:])

                backtracking(nums,i+1)
                path.pop()

        backtracking(nums,0)

        return res
