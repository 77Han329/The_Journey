class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []

        path = []

        def backtracking(n,k,start_idx):
            if len(path) == 3:
                if sum(path) == n:
                    res.append(path[:])
                return
            for i in range(start_idx,10):
                path.append(i)
                backtracking(n,k,i+1)
                path.pop()
        
        backtracking(n,k,1)
        return res
