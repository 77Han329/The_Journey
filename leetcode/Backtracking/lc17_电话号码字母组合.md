```python
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits:
            return []
        letter_map = [
            "",     # 0
            "",     # 1
            "abc",  # 2
            "def",  # 3
            "ghi",  # 4
            "jkl",  # 5
            "mno",  # 6
            "pqrs", # 7
            "tuv",  # 8
            "wxyz"  # 9
        ]

        path = []
        res = []

        def backtracking(index):
            
            if (len(path))==len(digits):
                res.append("".join(path))
                return
            
            cur_digit = int(digits[index])
            letters = letter_map[cur_digit]

            for ch in letters:
                path.append(ch)
                backtracking(index+1)
                path.pop()

        backtracking(0)
        return res

```
