class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        
        nums.sort()
        res = []
        path = []

        res.append(path)

        def backtracking(nums, start_idx):

            if start_idx >=len(nums):
                return 
            
            for i in range(start_idx,len(nums)):
                if i > start_idx and nums[i]==nums[i-1]:
                    continue
                path.append(nums[i])
                res.append(path[:])

                backtracking(nums,i+1)
                path.pop()

        backtracking(nums,0)

        return res
