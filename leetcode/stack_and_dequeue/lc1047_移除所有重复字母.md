class Solution:
    def removeDuplicates(self, s: str) -> str:
        """"
        Example:
        abbac
        [a] -> [a,b] -> (b = b) -> [a]-> (a,a) -> [] -> [c]     
         """"
        stack = []

        for item in s:
            if not stack:
                stack.append(item)
            else:
                if stack[-1] == item:
                    stack.pop()
                else: stack.append(item)

        return "".join(stack)
