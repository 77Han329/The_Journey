class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        for item in tokens:
            if item in ['+', '-', '*', '/'] and len(stack) >= 2:
                a = stack.pop()
                b = stack.pop()
                if item == '+':
                    c = a + b
                if item == '-':
                    c = b - a
                if item == '*':
                    c = a * b
                if item == '/':
                    # Python é™¤æ³•é»˜è®¤æ˜¯æµ®ç‚¹æ•°ï¼Œè¿™é‡Œè¦æˆªæ–­ä¸ºæ•´æ•°
                    c = int(b / a)
                stack.append(c)
            else:
                stack.append(int(item))
        return stack.pop()
